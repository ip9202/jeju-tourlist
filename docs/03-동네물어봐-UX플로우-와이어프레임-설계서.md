## 🗺️ 사용자 여정별 상세 플로우

### 1단계: 여행 전 (Pre-Trip) - 웹 중심 계획 단계

#### 🎯 사용자 목표
- 신뢰할 수 있는 여행 정보 수집
- 상세한 여행 계획 수립
- 여러 옵션 비교 및 검토
- 질문 리스트 작성 및 관리

#### 📊 상세 플로우

```mermaid
graph TD
    A[랜딩 페이지 접속] --> B{로그인 상태?}
    B -->|No| C[소셜 로그인]
    B -->|Yes| D[메인 대시보드]

    C --> D
    D --> E[여행 정보 입력]
    E --> F[맞춤 질문 추천]
    F --> G[질문 작성]
    G --> H[답변 대기]
    H --> I[답변 수신 알림]
    I --> J[답변 검토 및 채택]
    J --> K[여행 계획 노트 저장]
    K --> L[추가 질문 또는 완료]
```

#### 새 질문 등록 시 알림 플로우

```mermaid
sequenceDiagram
    participant U as 질문자
    participant S as 서버
    participant A as 답변자들
    participant N as 알림 시스템

    U->>S: 질문 등록
    S->>S: 태그/지역 기반 타겟 답변자 선별
    S->>N: 알림 대상자 리스트 전송

    N->>A: 실시간 웹 알림
    N->>A: 모바일 푸시 알림
    N->>A: 이메일 알림 (설정시)

    A->>S: 답변 작성 시작 (타이핑)
    S->>U: "답변 작성 중" 실시간 표시

    A->>S: 답변 등록
    S->>U: 새 답변 도착 알림
    S->>N: 다른 답변자들에게 경쟁 알림
```

const TypingIndicator = {

#### 실시간 해시태그 인식 및 링크

```javascript
// 해시태그 자동 파싱 및 스타일링
const HashtagParser = {
  // 입력 중 실시간 파싱
  parseOnInput: (text) => {
    const hashtagRegex = /#[\w가-힣]+/g;
    const hashtags = text.match(hashtagRegex) || [];

    return {
      originalText: text,
      hashtags: hashtags.map(tag => ({
        text: tag,
        category: this.categorizeTag(tag),
        suggestions: this.getSuggestions(tag)
      })),
      styledText: this.applyStyles(text, hashtags)
    };
  },

  // 태그 카테고리 분류
  categorizeTag: (tag) => {
    const categories = {
      region: ['제주시', '서귀포', '애월', '한림', '성산', '중문'],
      food: ['맛집', '카페', '술집', '디저트', '현지음식'],
      activity: ['해변', '등산', '드라이브', '체험', '사진명소'],
      timing: ['지금당장', '내일', '이번주말', '평일', '저녁'],
      group: ['혼자', '커플', '가족', '친구', '아이동반']
    };

    for (const [category, keywords] of Object.entries(categories)) {
      if (keywords.some(keyword => tag.includes(keyword))) {
        return category;
      }
    }
    return 'other';
  },

  // 관련 태그 제안
  getSuggestions: (partialTag) => {
    const popularTags = [
      '#제주맛집', '#애월카페', '#한라산', '#성산일출봉',
      '#중문해변', '#서귀포시장', '#제주흑돼지', '#한라봉'
    ];

    return popularTags.filter(tag =>
      tag.toLowerCase().includes(partialTag.toLowerCase().replace('#', ''))
    );
  }
};
```

// 답변 채택 플로우

  // 길게 누르기로 채택 메뉴 표시
  onLongPress: (answerId, element) => {
    const menu = this.createAcceptanceMenu(answerId);
    this.showMenu(menu, element);

        </div>

          📤 공유하기
        </div>
      </div>

      const response = await fetch(`/api/answers/${answerId}/accept`, {

        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ accepted: true })
      });
